Future<List<Covid19CountryBase>> getCountries() async {
  var res = await _dio.get('/countries');
  var data = res.data;
  List<Covid19CountryBase> list = List();

  for (var i in data) {
    list.add(Covid19CountryBase(
        country: i['Country'], slug: i['Slug'], iso2: i['ISO2']));
  }

  return list;
}
